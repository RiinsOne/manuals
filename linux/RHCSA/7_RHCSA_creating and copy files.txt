# создание множество файлов
touch file{5..15}

# тоже самое и для директорий
mkdir dir_{a..f}

# копирование с отображением (verbose)
cp -v file newfile

# копирование с подтверждением
cp -i file* dir_d

# не перезаписывать файлы
cp -vn file dir_a

# пропустить совпадающие файлы и скопировать новее или отсутствующие
cp -vu file* dir_b

# копирование с созданием жесткой ссылки
cp -vl file filelink
stat file filelink  # иноды будут совпадать

# копирование директории
cp -vr dir_a dir_b

# копирование с сохранием прав и владельцев у копии
# -a похоже на -rd
cp -a dir_a dir_c
