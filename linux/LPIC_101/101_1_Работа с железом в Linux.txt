# root права
sudo -s

# папка sys - куда монтируется файловая система, виртуальная sysfs
cd /sys

# block - блочные устройства
ls /sys/block

# bus - шины (процессоры, pci, usb и т.д.)

# устройства usb
ls bus/usb/devices/

# /proc - все запущенные процессы

# cpuinfo - информация по процессору
cd /proc
cat cpuinfo

# mounts - все смонтированные файловые системы
cat /proc/mounts

# file-max - сколько файлов могут быть открыто одновременно
cat /proc/sys/fs/file-max

# поменять значение через echo
echo 1000000 > file-max
cat file-max

# после перезагрузки измнения обнулятся

# dev - информация об устройствах
# интерфейсы работающие с драйвером ядра

# lsmod - информация о модулях ядра
# lspci - информация о PCI устройствах
# lspcmcia - информация об устройствах PCMCIA (ноутбуки)
# lsub - информация о шине USB
# lshal - база устройств HAL
# lshw - детальная информация комплектующих

# утилиты работы с модулем ядра
# lsmod
# modinfo - информация о конкретном модуле
# rmmod - удаление модуля ядра
# insmod - установка модуля ядра
# modprobe - деликатное удаление или добавление модулей

# сетевой драйвер
modinfo e1000
rmmod e1000  # удалить модуль
modprobe e1000  # аккуратно установить модуль сетевого драйвера

# IRQ - механизм прерываний
# IO адреса - обмен информацией между устройствами и CPU
# DMA - обращение к ОЗУ минуя CPU
cat /proc/interrupts  # прерывания
cat /proc/ioports  # адресация io по портам
cat /proc/iomem  # адресация в памяти
cat /proc/dma

# PATA - параллельный интерфейс
# SATA - последовательный интерфейс
# SCSI - стандарт передачи данных
# SAS - замена SCSI

# какие устройства куда смонтированы
blkid
